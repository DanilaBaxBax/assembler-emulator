.data  # секция данных
0    # mem[0] место для записи суммы элементов массива
10   # mem[1] длина массива
-10  # mem[2], здесь и далее - элементы массива
9
8
7
6
5
4
3
2
1
.code  # секция программы (сумма элементов массива)
          mov RI 1 0    # загрузка 0 в reg[1] : сумма
          mov RM 2 1    # загрузка mem[1] в reg[2] : счетчик цикла прохода по массиву
          jmp outlen    # переход к метке outlen (для вывода длины массива)
initloop: mov RI 3 2    # загрузка 2 в reg[3] : указатель на начало массива (mem[2])
loop:     add RX 1 3    # reg[1] = reg[1] + mem[reg[3]]
          sub RI 2 1    # reg[2] = reg[2] - 1
          add RI 3 1    # reg[3] = reg[3] + 1
          cmp RI 2 0    # сравнение reg[2] с 0
          jg  loop      # если результат сравнения >, переход к метке loop
          mov MR 0 1    # запись суммы элементов массива reg[1] в mem[0]
          out M 0       # вывод суммы элементов из mem[0]
          halt          # останов
outlen:   out R 2       # вывод длины массива из reg[2]
          jmp initloop  # переход к метке initloop для продожения работы программы
